–Т–љ–µ–і—А–µ–љ–Є–µ —А–µ—И–µ–љ–Є–є –і–ї—П —Г–њ—А–∞–≤–ї–µ–љ–Є—П –ґ—Г—А–љ–∞–ї–∞–Љ–Є –Є –Є—Б–њ–Њ–ї—М–Ј–Њ–≤–∞–љ–Є–µ —Б—А–µ–і—Б—В–≤ –Љ–Њ–љ–Є—В–Њ—А–Є–љ–≥–∞ –њ–Њ–Ј–≤–Њ–ї—П–µ—В –Њ–њ–µ—А–µ–ґ–∞—В—М –њ–Њ—В–µ–љ—Ж–Є–∞–ї—М–љ—Л–µ —Г–≥—А–Њ–Ј—Л –±–µ–Ј–Њ–њ–∞—Б–љ–Њ—Б—В–Є –Є —Н—Д—Д–µ–Ї—В–Є–≤–љ–Њ –Ј–∞—Й–Є—Й–∞—В—М –њ—А–Є–ї–Њ–ґ–µ–љ–Є—П Python.

8. –Ю–±—Г—З–µ–љ–Є–µ –Є –њ–Њ–і–≥–Њ—В–Њ–≤–Ї–∞ —А–∞–Ј—А–∞–±–Њ—В—З–Є–Ї–Њ–≤ #

–І—В–Њ–±—Л —Г–Ї—А–µ–њ–Є—В—М –ї—Г—З—И–Є–µ –њ—А–∞–Ї—В–Є–Ї–Є –±–µ–Ј–Њ–њ–∞—Б–љ–Њ—Б—В–Є Python, –Њ—З–µ–љ—М –≤–∞–ґ–љ–Њ –≤–Ї–ї–∞–і—Л–≤–∞—В—М —Б—А–µ–і—Б—В–≤–∞ –≤ –Њ–±—Г—З–µ–љ–Є–µ –Є –њ–Њ–і–≥–Њ—В–Њ–≤–Ї—Г —А–∞–Ј—А–∞–±–Њ—В—З–Є–Ї–Њ–≤ Python. –Я—А–µ–і–Њ—Б—В–∞–≤–ї—П—П –Є–Љ –љ–µ–Њ–±—Е–Њ–і–Є–Љ—Л–µ –Ј–љ–∞–љ–Є—П –Є –љ–∞–≤—Л–Ї–Є, –≤—Л –і–∞–µ—В–µ –≤–Њ–Ј–Љ–Њ–ґ–љ–Њ—Б—В—М —Б–≤–Њ–µ–є –Ї–Њ–Љ–∞–љ–і–µ —А–∞–Ј—А–∞–±–Њ—В—З–Є–Ї–Њ–≤ –њ–Є—Б–∞—В—М –±–µ–Ј–Њ–њ–∞—Б–љ—Л–є –Ї–Њ–і –Є –Њ–±–љ–∞—А—Г–ґ–Є–≤–∞—В—М –њ–Њ—В–µ–љ—Ж–Є–∞–ї—М–љ—Л–µ –њ—А–Њ–±–ї–µ–Љ—Л –±–µ–Ј–Њ–њ–∞—Б–љ–Њ—Б—В–Є –љ–∞ —А–∞–љ–љ–Є—Е —Н—В–∞–њ–∞—Е –ґ–Є–Ј–љ–µ–љ–љ–Њ–≥–Њ —Ж–Є–Ї–ї–∞ —А–∞–Ј—А–∞–±–Њ—В–Ї–Є.

–Т–Њ—В –љ–µ–Ї–Њ—В–Њ—А—Л–µ —И–∞–≥–Є, –Ї–Њ—В–Њ—А—Л–µ –Љ–Њ–ґ–љ–Њ –њ—А–µ–і–њ—А–Є–љ—П—В—М –і–ї—П –њ–Њ–≤—Л—И–µ–љ–Є—П —Г—А–Њ–≤–љ—П –Њ–±—А–∞–Ј–Њ–≤–∞–љ–Є—П –Є –њ–Њ–і–≥–Њ—В–Њ–≤–Ї–Є —А–∞–Ј—А–∞–±–Њ—В—З–Є–Ї–Њ–≤:

  • –Я—А–Њ–≥—А–∞–Љ–Љ—Л –њ–Њ–≤—Л—И–µ–љ–Є—П –Њ—Б–≤–µ–і–Њ–Љ–ї–µ–љ–љ–Њ—Б—В–Є –Њ –±–µ–Ј–Њ–њ–∞—Б–љ–Њ—Б—В–Є: –†–µ–≥—Г–ї—П—А–љ–Њ –њ—А–Њ–≤–Њ–і–Є—В–µ –њ—А–Њ–≥—А–∞–Љ–Љ—Л –њ–Њ–≤—Л—И–µ–љ–Є—П –Њ—Б–≤–µ–і–Њ–Љ–ї–µ–љ–љ–Њ—Б—В–Є –Њ –±–µ–Ј–Њ–њ–∞—Б–љ–Њ—Б—В–Є, —З—В–Њ–±—Л –Њ–Ј–љ–∞–Ї–Њ–Љ–Є—В—М —А–∞–Ј—А–∞–±–Њ—В—З–Є–Ї–Њ–≤ —Б —А–∞—Б–њ—А–Њ—Б—В—А–∞–љ–µ–љ–љ—Л–Љ–Є —Г—П–Ј–≤–Є–Љ–Њ—Б—В—П–Љ–Є –Є –њ—А–Є–µ–Љ–∞–Љ–Є –±–µ–Ј–Њ–њ–∞—Б–љ–Њ–≥–Њ –Ї–Њ–і–Є—А–Њ–≤–∞–љ–Є—П. –≠—В–Є –њ—А–Њ–≥—А–∞–Љ–Љ—Л –Љ–Њ–≥—Г—В –≤–Ї–ї—О—З–∞—В—М –≤ —Б–µ–±—П —Б–µ–Љ–Є–љ–∞—А—Л, –≤–µ–±–Є–љ–∞—А—Л –Є–ї–Є –Њ–љ–ї–∞–є–љ-—В—А–µ–љ–Є–љ–≥–Є, –Њ—А–Є–µ–љ—В–Є—А–Њ–≤–∞–љ–љ—Л–µ –љ–∞ —А–∞–Ј—А–∞–±–Њ—В–Ї—Г –њ—А–Є–ї–Њ–ґ–µ–љ–Є–є Python.

  • –†—Г–Ї–Њ–≤–Њ–і—Б—В–≤–Њ –њ–Њ –±–µ–Ј–Њ–њ–∞—Б–љ–Њ–Љ—Г –Ї–Њ–і–Є—А–Њ–≤–∞–љ–Є—О: –†–∞–Ј—А–∞–±–Њ—В–Ї–∞ —А–µ–Ї–Њ–Љ–µ–љ–і–∞—Ж–Є–є –њ–Њ –±–µ–Ј–Њ–њ–∞—Б–љ–Њ–Љ—Г –Ї–Њ–і–Є—А–Њ–≤–∞–љ–Є—О –і–ї—П —А–∞–Ј—А–∞–±–Њ—В–Ї–Є –љ–∞ —П–Ј—Л–Ї–µ Python —Б —Г–Ї–∞–Ј–∞–љ–Є–µ–Љ —А–µ–Ї–Њ–Љ–µ–љ–і—Г–µ–Љ—Л—Е –Љ–µ—В–Њ–і–Њ–≤ –Є –Љ–Њ–і–µ–ї–µ–є –Ї–Њ–і–∞, –њ–Њ–Ј–≤–Њ–ї—П—О—Й–Є—Е —Г—Б—В—А–∞–љ–Є—В—М —А–∞—Б–њ—А–Њ—Б—В—А–∞–љ–µ–љ–љ—Л–µ —Г—П–Ј–≤–Є–Љ–Њ—Б—В–Є. –≠—В–Є —А–µ–Ї–Њ–Љ–µ–љ–і–∞—Ж–Є–Є –Љ–Њ–≥—Г—В –Њ—Е–≤–∞—В—Л–≤–∞—В—М —В–∞–Ї–Є–µ —В–µ–Љ—Л, –Ї–∞–Ї –њ—А–Њ–≤–µ—А–Ї–∞ –≤–≤–Њ–і–Є–Љ—Л—Е –і–∞–љ–љ—Л—Е, –±–µ–Ј–Њ–њ–∞—Б–љ–∞—П –∞—Г—В–µ–љ—В–Є—Д–Є–Ї–∞—Ж–Є—П, —И–Є—Д—А–Њ–≤–∞–љ–Є–µ –і–∞–љ–љ—Л—Е –Є –±–µ–Ј–Њ–њ–∞—Б–љ–∞—П —А–∞–±–Њ—В–∞ —Б –Ї–Њ–љ—Д–Є–і–µ–љ—Ж–Є–∞–ї—М–љ–Њ–є –Є–љ—Д–Њ—А–Љ–∞—Ж–Є–µ–є.
