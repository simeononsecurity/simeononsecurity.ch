–Я—А–Њ—В–Њ–Ї–Њ–ї—Л –і–Є–љ–∞–Љ–Є—З–µ—Б–Ї–Њ–є –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є –њ—А–µ–і–љ–∞–Ј–љ–∞—З–µ–љ—Л –і–ї—П –∞–≤—В–Њ–Љ–∞—В–Є–Ј–∞—Ж–Є–Є –њ—А–Њ—Ж–µ—Б—Б–∞ –Њ–±–Љ–µ–љ–∞ –Є–љ—Д–Њ—А–Љ–∞—Ж–Є–µ–є –Њ –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є –Љ–µ–ґ–і—Г –Љ–∞—А—И—А—Г—В–Є–Ј–∞—В–Њ—А–∞–Љ–Є. –Ю–љ–Є –∞–і–∞–њ—В–Є—А—Г—О—В—Б—П –Ї –Є–Ј–Љ–µ–љ–µ–љ–Є—П–Љ –≤ —Б–µ—В–Є, —В–∞–Ї–Є–Љ –Ї–∞–Ї –Љ–Њ–і–Є—Д–Є–Ї–∞—Ж–Є—П —В–Њ–њ–Њ–ї–Њ–≥–Є–Є –Є–ї–Є –Њ—В–Ї–∞–Ј –Ї–∞–љ–∞–ї–Њ–≤ —Б–≤—П–Ј–Є, –њ—Г—В–µ–Љ –і–Є–љ–∞–Љ–Є—З–µ—Б–Ї–Њ–≥–Њ –Њ–±–љ–Њ–≤–ї–µ–љ–Є—П —В–∞–±–ї–Є—Ж –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є. –†–∞—Б—Б–Љ–Њ—В—А–Є–Љ –њ–Њ–і—А–Њ–±–љ–µ–µ –љ–µ–Ї–Њ—В–Њ—А—Л–µ —З–∞—Б—В–Њ –Є—Б–њ–Њ–ї—М–Ј—Г–µ–Љ—Л–µ –њ—А–Њ—В–Њ–Ї–Њ–ї—Л –і–Є–љ–∞–Љ–Є—З–µ—Б–Ї–Њ–є –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є:

–Я—А–Њ—В–Њ–Ї–Њ–ї –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є –Ш–љ—В–µ—А–љ–µ—В–∞ (RIP) #

–Я—А–Њ—В–Њ–Ї–Њ–ї Routing Internet Protocol (RIP) - —Н—В–Њ –њ—А–Њ—В–Њ–Ї–Њ–ї –≤–µ–Ї—В–Њ—А–љ–Њ–є –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є, —А–∞–±–Њ—В–∞ –Ї–Њ—В–Њ—А–Њ–≥–Њ –Њ—Б–љ–Њ–≤–∞–љ–∞ –љ–∞ –Ї–Њ–ї–Є—З–µ—Б—В–≤–µ –њ–µ—А–µ—Е–Њ–і–Њ–≤ –Љ–µ–ґ–і—Г –Љ–∞—А—И—А—Г—В–Є–Ј–∞—В–Њ—А–∞–Љ–Є. –Ю–љ –Є—Б–њ–Њ–ї—М–Ј—Г–µ—В –Љ–µ—В—А–Є–Ї—Г –Ї–Њ–ї–Є—З–µ—Б—В–≤–∞ —Е–Њ–њ–Њ–≤ –і–ї—П –Њ–њ—А–µ–і–µ–ї–µ–љ–Є—П –љ–∞–Є–ї—Г—З—И–µ–≥–Њ –њ—Г—В–Є –Ї —Б–µ—В–Є –љ–∞–Ј–љ–∞—З–µ–љ–Є—П. RIP –њ–Њ–і–і–µ—А–ґ–Є–≤–∞–µ—В –њ—А–Њ—В–Њ–Ї–Њ–ї—Л IPv4 –Є IPv6 –Є –њ–Њ–і—Е–Њ–і–Є—В –і–ї—П —Б–µ—В–µ–є –Љ–∞–ї–Њ–≥–Њ –Є —Б—А–µ–і–љ–µ–≥–Њ —А–∞–Ј–Љ–µ—А–∞.

–Ю—В–Ї—А—Л—В—Л–є –Ї—А–∞—В—З–∞–є—И–Є–є –њ—Г—В—М (Open Shortest Path First, OSPF) #

Open Shortest Path First (OSPF) - —Н—В–Њ –њ—А–Њ—В–Њ–Ї–Њ–ї –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є —Б –Ї–Њ–љ—В—А–Њ–ї–µ–Љ —Б–Њ—Б—В–Њ—П–љ–Є—П –Ї–∞–љ–∞–ї–Њ–≤, –Ї–Њ—В–Њ—А—Л–є –≤—Л—З–Є—Б–ї—П–µ—В –Ї—А–∞—В—З–∞–є—И–Є–є –њ—Г—В—М –Ї –Љ–µ—Б—В—Г –љ–∞–Ј–љ–∞—З–µ–љ–Є—П —Б –њ–Њ–Љ–Њ—Й—М—О –∞–ї–≥–Њ—А–Є—В–Љ–∞ –Ф–µ–є–Ї—Б—В—А—Л. –Ф–ї—П –Њ–њ—А–µ–і–µ–ї–µ–љ–Є—П –Њ–њ—В–Є–Љ–∞–ї—М–љ–Њ–≥–Њ –Љ–∞—А—И—А—Г—В–∞ –Њ–љ —Г—З–Є—В—Л–≤–∞–µ—В —А–∞–Ј–ї–Є—З–љ—Л–µ –Љ–µ—В—А–Є–Ї–Є, —В–∞–Ї–Є–µ –Ї–∞–Ї –њ—А–Њ–њ—Г—Б–Ї–љ–∞—П —Б–њ–Њ—Б–Њ–±–љ–Њ—Б—В—М, –Ј–∞–і–µ—А–ґ–Ї–∞ –Є –љ–∞–і–µ–ґ–љ–Њ—Б—В—М. OSPF —И–Є—А–Њ–Ї–Њ –Є—Б–њ–Њ–ї—М–Ј—Г–µ—В—Б—П –≤ –Ї—А—Г–њ–љ—Л—Е –Ї–Њ—А–њ–Њ—А–∞—В–Є–≤–љ—Л—Е —Б–µ—В—П—Е –±–ї–∞–≥–Њ–і–∞—А—П —Б–≤–Њ–µ–є –Љ–∞—Б—И—В–∞–±–Є—А—Г–µ–Љ–Њ—Б—В–Є –Є –±—Л—Б—В—А–Њ–є —Б—Е–Њ–і–Є–Љ–Њ—Б—В–Є.

Enhanced Interior Gateway Routing Protocol (EIGRP) #

Enhanced Interior Gateway Routing Protocol (EIGRP) - —Н—В–Њ –≥–Є–±—А–Є–і–љ—Л–є –њ—А–Њ—В–Њ–Ї–Њ–ї –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є, —А–∞–Ј—А–∞–±–Њ—В–∞–љ–љ—Л–є –Ї–Њ–Љ–њ–∞–љ–Є–µ–є Cisco. –Ю–љ —Б–Њ—З–µ—В–∞–µ—В –≤ —Б–µ–±–µ –ї—Г—З—И–Є–µ —З–µ—А—В—Л –њ—А–Њ—В–Њ–Ї–Њ–ї–Њ–≤ —Б –≤–µ–Ї—В–Њ—А–Њ–Љ —А–∞—Б—Б—В–Њ—П–љ–Є—П –Є –њ—А–Њ—В–Њ–Ї–Њ–ї–Њ–≤ —Б —Б–Њ—Б—В–Њ—П–љ–Є–µ–Љ –Ї–∞–љ–∞–ї–Њ–≤. EIGRP –Є—Б–њ–Њ–ї—М–Ј—Г–µ—В –∞–ї–≥–Њ—А–Є—В–Љ –і–Є—Д—Д—Г–Ј–љ–Њ–≥–Њ –Њ–±–љ–Њ–≤–ї–µ–љ–Є—П (DUAL) –і–ї—П —А–∞—Б—З–µ—В–∞ –Љ–∞—А—И—А—Г—В–Њ–≤ –Є –њ—А–µ–і–Њ—Б—В–∞–≤–ї—П–µ—В —В–∞–Ї–Є–µ —А–∞—Б—И–Є—А–µ–љ–љ—Л–µ –≤–Њ–Ј–Љ–Њ–ґ–љ–Њ—Б—В–Є, –Ї–∞–Ї –±–∞–ї–∞–љ—Б–Є—А–Њ–≤–Ї–∞ –љ–∞–≥—А—Г–Ј–Ї–Є –њ–Њ –љ–µ—А–∞–≤–љ—Л–Љ –Ј–∞—В—А–∞—В–∞–Љ –Є —Б—Г–Љ–Љ–Є—А–Њ–≤–∞–љ–Є–µ –Љ–∞—А—И—А—Г—В–Њ–≤.

–Я—А–Њ—В–Њ–Ї–Њ–ї –њ–Њ–≥—А–∞–љ–Є—З–љ–Њ–≥–Њ —И–ї—О–Ј–∞ (BGP) #

Border Gateway Protocol (BGP) - —Н—В–Њ –њ—А–Њ—В–Њ–Ї–Њ–ї –≤–љ–µ—И–љ–µ–≥–Њ —И–ї—О–Ј–∞, –Є—Б–њ–Њ–ї—М–Ј—Г–µ–Љ—Л–є –і–ї—П –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є –Љ–µ–ґ–і—Г –∞–≤—В–Њ–љ–Њ–Љ–љ—Л–Љ–Є —Б–Є—Б—В–µ–Љ–∞–Љ–Є (ASes) –≤ –Ш–љ—В–µ—А–љ–µ—В–µ. BGP –Њ–±–ї–∞–і–∞–µ—В –≤—Л—Б–Њ–Ї–Њ–є –Љ–∞—Б—И—В–∞–±–Є—А—Г–µ–Љ–Њ—Б—В—М—О –Є –њ–Њ–Ј–≤–Њ–ї—П–µ—В –∞–≤—В–Њ–љ–Њ–Љ–љ—Л–Љ —Б–Є—Б—В–µ–Љ–∞–Љ –Њ–±–Љ–µ–љ–Є–≤–∞—В—М—Б—П –Є–љ—Д–Њ—А–Љ–∞—Ж–Є–µ–є –Њ –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є. –Ю–љ –Є—Б–њ–Њ–ї—М–Ј—Г–µ—В –∞—В—А–Є–±—Г—В—Л –њ—Г—В–Є –Є –њ–Њ–ї–Є—В–Є–Ї–Є –і–ї—П –њ—А–Є–љ—П—В–Є—П —А–µ—И–µ–љ–Є–є –Њ –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є –љ–∞ –Њ—Б–љ–Њ–≤–µ —В–∞–Ї–Є—Е —Д–∞–Ї—В–Њ—А–Њ–≤, –Ї–∞–Ї —Б–µ—В–µ–≤–∞—П –њ–Њ–ї–Є—В–Є–Ї–∞, –і–ї–Є–љ–∞ –њ—Г—В–Є –Є –њ—Г—В—М AS.

–Я–Њ –њ—А–Є–љ—Ж–Є–њ—Г —А–∞–±–Њ—В—Л –Є –Є–љ—Д–Њ—А–Љ–∞—Ж–Є–Є, –Є—Б–њ–Њ–ї—М–Ј—Г–µ–Љ–Њ–є –і–ї—П –Њ–њ—А–µ–і–µ–ї–µ–љ–Є—П –Љ–∞—А—И—А—Г—В–Њ–≤, –њ—А–Њ—В–Њ–Ї–Њ–ї—Л –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є –Љ–Њ–ґ–љ–Њ —А–∞–Ј–і–µ–ї–Є—В—М –љ–∞ –њ—А–Њ—В–Њ–Ї–Њ–ї—Л —Б —Б–Њ—Б—В–Њ—П–љ–Є–µ–Љ –Ї–∞–љ–∞–ї–∞, —Б –≤–µ–Ї—В–Њ—А–Њ–Љ —А–∞—Б—Б—В–Њ—П–љ–Є—П –Є –≥–Є–±—А–Є–і–љ—Л–µ.

–Я—А–Њ—В–Њ–Ї–Њ–ї—Л –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є —Б —Б–Њ—Б—В–Њ—П–љ–Є–µ–Љ –Ї–∞–љ–∞–ї–∞ #

–Я—А–Њ—В–Њ–Ї–Њ–ї—Л –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є —Б –Ї–Њ–љ—В—А–Њ–ї–µ–Љ —Б–Њ—Б—В–Њ—П–љ–Є—П –Ї–∞–љ–∞–ї–Њ–≤, —В–∞–Ї–Є–µ –Ї–∞–Ї OSPF, –њ–Њ–і–і–µ—А–ґ–Є–≤–∞—О—В –њ–Њ–і—А–Њ–±–љ—Г—О –Ї–∞—А—В—Г –≤—Б–µ–є —Б–µ—В–Є, –Њ–±–Љ–µ–љ–Є–≤–∞—П—Б—М –Є–љ—Д–Њ—А–Љ–∞—Ж–Є–µ–є –Њ —Б–Њ—Б—В–Њ—П–љ–Є–Є –Ї–∞–љ–∞–ї–Њ–≤ –Љ–µ–ґ–і—Г –Љ–∞—А—И—А—Г—В–Є–Ј–∞—В–Њ—А–∞–Љ–Є. –Ъ–∞–ґ–і—Л–є –Љ–∞—А—И—А—Г—В–Є–Ј–∞—В–Њ—А —Б—В—А–Њ–Є—В —В–Њ–њ–Њ–ї–Њ–≥–Є—З–µ—Б–Ї—Г—О –±–∞–Ј—Г –і–∞–љ–љ—Л—Е, –њ–Њ–Ј–≤–Њ–ї—П—О—Й—Г—О –µ–Љ—Г —А–∞—Б—Б—З–Є—В—Л–≤–∞—В—М –љ–∞–Є–ї—Г—З—И–Є–є –њ—Г—В—М –Ї —Б–µ—В–Є –љ–∞–Ј–љ–∞—З–µ–љ–Є—П –љ–∞ –Њ—Б–љ–Њ–≤–µ —А–∞–Ј–ї–Є—З–љ—Л—Е –Љ–µ—В—А–Є–Ї.

–Я—А–Њ—В–Њ–Ї–Њ–ї—Л –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є —Б –≤–µ–Ї—В–Њ—А–Њ–Љ —А–∞—Б—Б—В–Њ—П–љ–Є—П #

–Я—А–Њ—В–Њ–Ї–Њ–ї—Л –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є —Б –≤–µ–Ї—В–Њ—А–Њ–Љ —А–∞—Б—Б—В–Њ—П–љ–Є—П, —В–∞–Ї–Є–µ –Ї–∞–Ї RIP, –Є—Б–њ–Њ–ї—М–Ј—Г—О—В –њ—А–Њ—Б—В—Г—О –Љ–µ—В—А–Є–Ї—Г (–љ–∞–њ—А–Є–Љ–µ—А, —Б—З–µ—В—З–Є–Ї —Е–Њ–њ–Њ–≤) –Є –Њ–±–Љ–µ–љ–Є–≤–∞—О—В—Б—П –Є–љ—Д–Њ—А–Љ–∞—Ж–Є–µ–є –Њ –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є —Б —Б–Њ—Б–µ–і–љ–Є–Љ–Є –Љ–∞—А—И—А—Г—В–Є–Ј–∞—В–Њ—А–∞–Љ–Є. –Ь–∞—А—И—А—Г—В–Є–Ј–∞—В–Њ—А—Л –њ–µ—А–Є–Њ–і–Є—З–µ—Б–Ї–Є —А–µ–Ї–ї–∞–Љ–Є—А—Г—О—В —Б–≤–Њ–Є —В–∞–±–ї–Є—Ж—Л –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є —Б–Њ—Б–µ–і–љ–Є–Љ –Љ–∞—А—И—А—Г—В–Є–Ј–∞—В–Њ—А–∞–Љ, —З—В–Њ –њ–Њ–Ј–≤–Њ–ї—П–µ—В –Є–Љ —Б–Њ—Б—В–∞–≤–Є—В—М –њ—А–µ–і—Б—В–∞–≤–ї–µ–љ–Є–µ –Њ —Б–µ—В–Є. –Я—А–Њ—В–Њ–Ї–Њ–ї—Л –≤–µ–Ї—В–Њ—А–∞ —А–∞—Б—Б—В–Њ—П–љ–Є—П –Є–Љ–µ—О—В –Њ–≥—А–∞–љ–Є—З–µ–љ–љ–Њ–µ –њ—А–µ–і—Б—В–∞–≤–ї–µ–љ–Є–µ –Њ —Б–µ—В–Є –≤ —Ж–µ–ї–Њ–Љ –Є –Љ–Њ–≥—Г—В –±—Л—В—М —Б–Ї–ї–Њ–љ–љ—Л –Ї –Њ–±—А–∞–Ј–Њ–≤–∞–љ–Є—О –њ–µ—В–µ–ї—М –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є.

–У–Є–±—А–Є–і–љ—Л–µ –њ—А–Њ—В–Њ–Ї–Њ–ї—Л –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є #

–У–Є–±—А–Є–і–љ—Л–µ –њ—А–Њ—В–Њ–Ї–Њ–ї—Л –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є, —В–∞–Ї–Є–µ –Ї–∞–Ї EIGRP, —Б–Њ—З–µ—В–∞—О—В –≤ —Б–µ–±–µ –≤–Њ–Ј–Љ–Њ–ґ–љ–Њ—Б—В–Є –њ—А–Њ—В–Њ–Ї–Њ–ї–Њ–≤ —Б —Б–Њ—Б—В–Њ—П–љ–Є–µ–Љ –Ї–∞–љ–∞–ї–Њ–≤ –Є –њ—А–Њ—В–Њ–Ї–Њ–ї–Њ–≤ —Б –≤–µ–Ї—В–Њ—А–Њ–Љ —А–∞—Б—Б—В–Њ—П–љ–Є—П. –Ю–љ–Є –≤–µ–і—Г—В —В–∞–±–ї–Є—Ж—Г —В–Њ–њ–Њ–ї–Њ–≥–Є–Є, –Ї–∞–Ї –њ—А–Њ—В–Њ–Ї–Њ–ї—Л —Б–Њ—Б—В–Њ—П–љ–Є—П —Б–≤—П–Ј–µ–є, –љ–Њ –Є—Б–њ–Њ–ї—М–Ј—Г—О—В –∞–ї–≥–Њ—А–Є—В–Љ—Л –≤–µ–Ї—В–Њ—А–∞ —А–∞—Б—Б—В–Њ—П–љ–Є–є –і–ї—П —А–∞—Б—З–µ—В–∞ –Љ–∞—А—И—А—Г—В–Њ–≤. –Я—А–µ–Є–Љ—Г—Й–µ—Б—В–≤–∞ –≥–Є–±—А–Є–і–љ—Л—Е –њ—А–Њ—В–Њ–Ї–Њ–ї–Њ–≤ –Ј–∞–Ї–ї—О—З–∞—О—В—Б—П –≤ –±–Њ–ї–µ–µ –±—Л—Б—В—А–Њ–є —Б—Е–Њ–і–Є–Љ–Њ—Б—В–Є –Є —Б–љ–Є–ґ–µ–љ–Є–Є –љ–∞–Ї–ї–∞–і–љ—Л—Е —А–∞—Б—Е–Њ–і–Њ–≤.

–°—В–∞—В–Є—З–µ—Б–Ї–∞—П –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є—П –Є –Љ–∞—А—И—А—Г—В—Л –њ–Њ —Г–Љ–Њ–ї—З–∞–љ–Є—О #

–°—В–∞—В–Є—З–µ—Б–Ї–∞—П –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є—П –њ—А–µ–і–њ–Њ–ї–∞–≥–∞–µ—В —А—Г—З–љ–Њ–µ –Ї–Њ–љ—Д–Є–≥—Г—А–Є—А–Њ–≤–∞–љ–Є–µ —В–∞–±–ї–Є—Ж—Л –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є –љ–∞ –Љ–∞—А—И—А—Г—В–Є–Ј–∞—В–Њ—А–∞—Е —Б —Г–Ї–∞–Ј–∞–љ–Є–µ–Љ –њ—Г—В–µ–є –і–ї—П –і–Њ—Б—В–Є–ґ–µ–љ–Є—П –Њ–њ—А–µ–і–µ–ї–µ–љ–љ—Л—Е —Б–µ—В–µ–є. –Ю–љ–∞ –Њ–±—Л—З–љ–Њ –Є—Б–њ–Њ–ї—М–Ј—Г–µ—В—Б—П –≤ —Б—Ж–µ–љ–∞—А–Є—П—Е, –≥–і–µ –Є–Ј–Љ–µ–љ–µ–љ–Є—П —В–Њ–њ–Њ–ї–Њ–≥–Є–Є —Б–µ—В–Є –Љ–Є–љ–Є–Љ–∞–ї—М–љ—Л –Є–ї–Є –њ—А–µ–і—Б–Ї–∞–Ј—Г–µ–Љ—Л. –°—В–∞—В–Є—З–µ—Б–Ї–Є–µ –Љ–∞—А—И—А—Г—В—Л –њ—А–Њ—Б—В—Л –≤ –љ–∞—Б—В—А–Њ–є–Ї–µ –Є –Љ–Њ–≥—Г—В –±—Л—В—М –њ–Њ–ї–µ–Ј–љ—Л –і–ї—П –љ–µ–±–Њ–ї—М—И–Є—Е —Б–µ—В–µ–є –Є–ї–Є –Њ—В–і–µ–ї—М–љ—Л—Е —Б–µ–≥–Љ–µ–љ—В–Њ–≤ —Б–µ—В–Є.

–Ь–∞—А—И—А—Г—В –њ–Њ —Г–Љ–Њ–ї—З–∞–љ–Є—О, —В–∞–Ї–ґ–µ –Є–Ј–≤–µ—Б—В–љ—Л–є –Ї–∞–Ї —И–ї—О–Ј –њ–Њ—Б–ї–µ–і–љ–µ–є –Є–љ—Б—В–∞–љ—Ж–Є–Є, –Є—Б–њ–Њ–ї—М–Ј—Г–µ—В—Б—П, –Ї–Њ–≥–і–∞ –і–ї—П —Б–µ—В–Є –љ–∞–Ј–љ–∞—З–µ–љ–Є—П –љ–µ —Б—Г—Й–µ—Б—В–≤—Г–µ—В —П–≤–љ–Њ–≥–Њ –Љ–∞—А—И—А—Г—В–∞. –Ю–љ –і–µ–є—Б—В–≤—Г–µ—В –Ї–∞–Ї —Г–љ–Є–≤–µ—А—Б–∞–ї—М–љ—Л–є –Љ–∞—А—И—А—Г—В –Є –љ–∞—Б—В—А–∞–Є–≤–∞–µ—В—Б—П –љ–∞ –Љ–∞—А—И—А—Г—В–Є–Ј–∞—В–Њ—А–∞—Е –і–ї—П –љ–∞–њ—А–∞–≤–ї–µ–љ–Є—П —В—А–∞—Д–Є–Ї–∞ –љ–∞ —И–ї—О–Ј –њ–Њ —Г–Љ–Њ–ї—З–∞–љ–Є—О, –Ї–Њ–≥–і–∞ –Љ–∞—А—И—А—Г—В–Є–Ј–∞—В–Њ—А –љ–µ –Є–Љ–µ–µ—В –Є–љ—Д–Њ—А–Љ–∞—Ж–Є–Є –Њ —Б–µ—В–Є –љ–∞–Ј–љ–∞—З–µ–љ–Є—П.

–Р–і–Љ–Є–љ–Є—Б—В—А–∞—В–Є–≤–љ–Њ–µ —А–∞—Б—Б—В–Њ—П–љ–Є–µ, –≤–љ–µ—И–љ–Є–µ –Є –≤–љ—Г—В—А–µ–љ–љ–Є–µ –Љ–∞—А—И—А—Г—В—Л –Є –≤—А–µ–Љ—П –ґ–Є–Ј–љ–Є #

–Р–і–Љ–Є–љ–Є—Б—В—А–∞—В–Є–≤–љ–Њ–µ —А–∞—Б—Б—В–Њ—П–љ–Є–µ #

–Р–і–Љ–Є–љ–Є—Б—В—А–∞—В–Є–≤–љ–Њ–µ —А–∞—Б—Б—В–Њ—П–љ–Є–µ (Administrative distance, AD) - —Н—В–Њ –Ј–љ–∞—З–µ–љ–Є–µ, –њ—А–Є—Б–≤–∞–Є–≤–∞–µ–Љ–Њ–µ –њ—А–Њ—В–Њ–Ї–Њ–ї–∞–Љ –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є –і–ї—П –Њ–њ—А–µ–і–µ–ї–µ–љ–Є—П –њ—А–Є–Њ—А–Є—В–µ—В–∞ –Љ–∞—А—И—А—Г—В–Њ–≤, –Ї–Њ–≥–і–∞ –љ–∞ –Љ–∞—А—И—А—Г—В–Є–Ј–∞—В–Њ—А–µ —А–∞–±–Њ—В–∞–µ—В –љ–µ—Б–Ї–Њ–ї—М–Ї–Њ –њ—А–Њ—В–Њ–Ї–Њ–ї–Њ–≤. –Ь–µ–љ—М—И–Є–µ –Ј–љ–∞—З–µ–љ–Є—П –∞–і–Љ–Є–љ–Є—Б—В—А–∞—В–Є–≤–љ–Њ–≥–Њ —А–∞—Б—Б—В–Њ—П–љ–Є—П —Г–Ї–∞–Ј—Л–≤–∞—О—В –љ–∞ –±–Њ–ї–µ–µ –≤—Л—Б–Њ–Ї–Є–є –њ—А–Є–Њ—А–Є—В–µ—В –Ї–Њ–љ–Ї—А–µ—В–љ–Њ–≥–Њ –њ—А–Њ—В–Њ–Ї–Њ–ї–∞ –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є. –Э–∞–њ—А–Є–Љ–µ—А, –њ—А–Њ—В–Њ–Ї–Њ–ї OSPF –Є–Љ–µ–µ—В –Љ–µ–љ—М—И–µ–µ –Ј–љ–∞—З–µ–љ–Є–µ AD (110), —З–µ–Љ –њ—А–Њ—В–Њ–Ї–Њ–ї RIP (120), –њ–Њ—Н—В–Њ–Љ—Г –Љ–∞—А—И—А—Г—В—Л OSPF –±—Г–і—Г—В –њ—А–µ–і–њ–Њ—З—В–Є—В–µ–ї—М–љ–µ–µ –Љ–∞—А—И—А—Г—В–Њ–≤ RIP.

–Т–љ–µ—И–љ—П—П –Є –≤–љ—Г—В—А–µ–љ–љ—П—П –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є—П #

–Т–љ–µ—И–љ–Є–µ –њ—А–Њ—В–Њ–Ї–Њ–ї—Л –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є, —В–∞–Ї–Є–µ –Ї–∞–Ї BGP, –Є—Б–њ–Њ–ї—М–Ј—Г—О—В—Б—П –і–ї—П –Њ–±–Љ–µ–љ–∞ –Є–љ—Д–Њ—А–Љ–∞—Ж–Є–µ–є –Њ –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є –Љ–µ–ґ–і—Г –∞–≤—В–Њ–љ–Њ–Љ–љ—Л–Љ–Є —Б–Є—Б—В–µ–Љ–∞–Љ–Є (ASes). –Ю–љ–Є –Њ–±–µ—Б–њ–µ—З–Є–≤–∞—О—В –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є—О –Љ–µ–ґ–і—Г —А–∞–Ј–ї–Є—З–љ—Л–Љ–Є –Њ—А–≥–∞–љ–Є–Ј–∞—Ж–Є—П–Љ–Є –Є –њ–Њ—Б—В–∞–≤—Й–Є–Ї–∞–Љ–Є —Г—Б–ї—Г–≥. –° –і—А—Г–≥–Њ–є —Б—В–Њ—А–Њ–љ—Л, –њ—А–Њ—В–Њ–Ї–Њ–ї—Л –≤–љ—Г—В—А–µ–љ–љ–µ–є –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є, —В–∞–Ї–Є–µ –Ї–∞–Ї RIP, OSPF –Є EIGRP, –Є—Б–њ–Њ–ї—М–Ј—Г—О—В—Б—П –і–ї—П –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є –≤–љ—Г—В—А–Є –∞–≤—В–Њ–љ–Њ–Љ–љ–Њ–є —Б–Є—Б—В–µ–Љ—Л.

–Т—А–µ–Љ—П –ґ–Є–Ј–љ–Є (TTL) #

–Т—А–µ–Љ—П –ґ–Є–Ј–љ–Є (TTL) - —Н—В–Њ –њ–Њ–ї–µ –≤ IP-–њ–∞–Ї–µ—В–∞—Е, –Њ–њ—А–µ–і–µ–ї—П—О—Й–µ–µ –Љ–∞–Ї—Б–Є–Љ–∞–ї—М–љ–Њ–µ –Ї–Њ–ї–Є—З–µ—Б—В–≤–Њ –њ–µ—А–µ—Е–Њ–і–Њ–≤, –Ї–Њ—В–Њ—А–Њ–µ –Љ–Њ–ґ–µ—В –њ—А–Њ–є—В–Є –њ–∞–Ї–µ—В –і–Њ —В–Њ–≥–Њ, –Ї–∞–Ї –±—Г–і–µ—В –Њ—В–±—А–Њ—И–µ–љ. –Ю–љ–Њ –љ–µ –њ–Њ–Ј–≤–Њ–ї—П–µ—В –њ–∞–Ї–µ—В–∞–Љ –±–µ—Б–Ї–Њ–љ–µ—З–љ–Њ —Ж–Є—А–Ї—Г–ї–Є—А–Њ–≤–∞—В—М –≤ —Б–µ—В–Є –њ—А–Є –љ–∞–ї–Є—З–Є–Є –њ–µ—В–µ–ї—М –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є –Є–ї–Є –і—А—Г–≥–Є—Е –њ—А–Њ–±–ї–µ–Љ. –Ъ–∞–ґ–і—Л–є –Љ–∞—А—И—А—Г—В–Є–Ј–∞—В–Њ—А —Г–Љ–µ–љ—М—И–∞–µ—В –Ј–љ–∞—З–µ–љ–Є–µ TTL –њ–Њ –Љ–µ—А–µ –њ—А–Њ—Е–Њ–ґ–і–µ–љ–Є—П –њ–∞–Ї–µ—В–∞ –њ–Њ —Б–µ—В–Є.

–£–њ—А–∞–≤–ї–µ–љ–Є–µ –њ—А–Њ–њ—Г—Б–Ї–љ–Њ–є —Б–њ–Њ—Б–Њ–±–љ–Њ—Б—В—М—О #

–≠—Д—Д–µ–Ї—В–Є–≤–љ–Њ–µ —Г–њ—А–∞–≤–ї–µ–љ–Є–µ –њ–Њ–ї–Њ—Б–Њ–є –њ—А–Њ–њ—Г—Б–Ї–∞–љ–Є—П –Є–Љ–µ–µ—В —А–µ—И–∞—О—Й–µ–µ –Ј–љ–∞—З–µ–љ–Є–µ –і–ї—П –Њ–њ—В–Є–Љ–Є–Ј–∞—Ж–Є–Є –њ—А–Њ–Є–Ј–≤–Њ–і–Є—В–µ–ї—М–љ–Њ—Б—В–Є —Б–µ—В–Є –Є –Њ–±–µ—Б–њ–µ—З–µ–љ–Є—П –±–µ—Б–њ–µ—А–µ–±–Њ–є–љ–Њ–є –њ–µ—А–µ–і–∞—З–Є —В—А–∞—Д–Є–Ї–∞. –Ф–≤—Г–Љ—П –≤–∞–ґ–љ—Л–Љ–Є –∞—Б–њ–µ–Ї—В–∞–Љ–Є —Г–њ—А–∞–≤–ї–µ–љ–Є—П –њ–Њ–ї–Њ—Б–Њ–є –њ—А–Њ–њ—Г—Б–Ї–∞–љ–Є—П —П–≤–ї—П—О—В—Б—П —Д–Њ—А–Љ–Є—А–Њ–≤–∞–љ–Є–µ —В—А–∞—Д–Є–Ї–∞ –Є –Ї–∞—З–µ—Б—В–≤–Њ –Њ–±—Б–ї—Г–ґ–Є–≤–∞–љ–Є—П (QoS).

–§–Њ—А–Љ–Є—А–Њ–≤–∞–љ–Є–µ —В—А–∞—Д–Є–Ї–∞ #

–§–Њ—А–Љ–Є—А–Њ–≤–∞–љ–Є–µ —В—А–∞—Д–Є–Ї–∞ - —Н—В–Њ —В–µ—Е–љ–Њ–ї–Њ–≥–Є—П, –њ–Њ–Ј–≤–Њ–ї—П—О—Й–∞—П –Ї–Њ–љ—В—А–Њ–ї–Є—А–Њ–≤–∞—В—М —Б–Ї–Њ—А–Њ—Б—В—М –њ–µ—А–µ–і–∞—З–Є –і–∞–љ–љ—Л—Е –≤ —Б–µ—В–Є. –Ю–љ–∞ –њ–Њ–Ј–≤–Њ–ї—П–µ—В —Б–µ—В–µ–≤—Л–Љ –∞–і–Љ–Є–љ–Є—Б—В—А–∞—В–Њ—А–∞–Љ —Д–Њ—А–Љ–Є—А–Њ–≤–∞—В—М –њ–Њ—В–Њ–Ї —В—А–∞—Д–Є–Ї–∞, –Њ–њ—А–µ–і–µ–ї—П—П –Њ–≥—А–∞–љ–Є—З–µ–љ–Є—П –њ—А–Њ–њ—Г—Б–Ї–љ–Њ–є —Б–њ–Њ—Б–Њ–±–љ–Њ—Б—В–Є –Є —Г—Б—В–∞–љ–∞–≤–ї–Є–≤–∞—П –њ—А–Є–Њ—А–Є—В–µ—В—Л –і–ї—П –Њ–њ—А–µ–і–µ–ї–µ–љ–љ—Л—Е —В–Є–њ–Њ–≤ —В—А–∞—Д–Є–Ї–∞. –≠—В–Њ –њ–Њ–Ј–≤–Њ–ї—П–µ—В –њ—А–µ–і–Њ—В–≤—А–∞—В–Є—В—М –њ–µ—А–µ–≥—А—Г–Ј–Ї—Г —Б–µ—В–Є –Є –Њ–±–µ—Б–њ–µ—З–Є—В—М –і–Њ—Б—В–∞—В–Њ—З–љ—Г—О –њ—А–Њ–њ—Г—Б–Ї–љ—Г—О —Б–њ–Њ—Б–Њ–±–љ–Њ—Б—В—М –і–ї—П –Ї—А–Є—В–Є—З–µ—Б–Ї–Є –≤–∞–ґ–љ—Л—Е –њ—А–Є–ї–Њ–ґ–µ–љ–Є–є.

–Ъ–∞—З–µ—Б—В–≤–Њ –Њ–±—Б–ї—Г–ґ–Є–≤–∞–љ–Є—П (QoS) #

–Ъ–∞—З–µ—Б—В–≤–Њ –Њ–±—Б–ї—Г–ґ–Є–≤–∞–љ–Є—П (QoS) - —Н—В–Њ —Б–њ–Њ—Б–Њ–±–љ–Њ—Б—В—М —Б–µ—В–Є —Г—Б—В–∞–љ–∞–≤–ї–Є–≤–∞—В—М –њ—А–Є–Њ—А–Є—В–µ—В—Л –Є –≤—Л–і–µ–ї—П—В—М —А–µ—Б—Г—А—Б—Л –і–ї—П —А–∞–Ј–ї–Є—З–љ—Л—Е —В–Є–њ–Њ–≤ —В—А–∞—Д–Є–Ї–∞ –≤ –Ј–∞–≤–Є—Б–Є–Љ–Њ—Б—В–Є –Њ—В –Є—Е –≤–∞–ґ–љ–Њ—Б—В–Є –Є —В—А–µ–±–Њ–≤–∞–љ–Є–є. –Ь–µ—Е–∞–љ–Є–Ј–Љ—Л QoS, —В–∞–Ї–Є–µ –Ї–∞–Ї –њ—А–Є–Њ—А–Є—В–µ–Ј–∞—Ж–Є—П —В—А–∞—Д–Є–Ї–∞, —А–∞—Б–њ—А–µ–і–µ–ї–µ–љ–Є–µ –њ–Њ–ї–Њ—Б—Л –њ—А–Њ–њ—Г—Б–Ї–∞–љ–Є—П –Є —Г–њ—А–∞–≤–ї–µ–љ–Є–µ –њ–µ—А–µ–≥—А—Г–Ј–Ї–∞–Љ–Є, –њ–Њ–Љ–Њ–≥–∞—О—В –Њ–±–µ—Б–њ–µ—З–Є—В—М –Њ–њ—В–Є–Љ–∞–ї—М–љ—Г—О –њ—А–Њ–Є–Ј–≤–Њ–і–Є—В–µ–ї—М–љ–Њ—Б—В—М –њ—А–Є–ї–Њ–ґ–µ–љ–Є–є —А–µ–∞–ї—М–љ–Њ–≥–Њ –≤—А–µ–Љ–µ–љ–Є, —В–∞–Ї–Є—Е –Ї–∞–Ї –≥–Њ–ї–Њ—Б –Є –≤–Є–і–µ–Њ.

–°—А–∞–≤–љ–µ–љ–Є–µ –Є —А–∞–Ј–Љ–µ—Й–µ–љ–Є–µ —Г—Б—В—А–Њ–є—Б—В–≤ #

–†–∞–Ј–ї–Є—З–љ—Л–µ —Г—Б—В—А–Њ–є—Б—В–≤–∞ –Є–≥—А–∞—О—В –Њ–њ—А–µ–і–µ–ї–µ–љ–љ—Л–µ —А–Њ–ї–Є –≤ —Б–µ—В–Є –Є –Њ–±–ї–∞–і–∞—О—В —А–∞–Ј–ї–Є—З–љ—Л–Љ–Є —Е–∞—А–∞–Ї—В–µ—А–Є—Б—В–Є–Ї–∞–Љ–Є, –њ–Њ–Ј–≤–Њ–ї—П—О—Й–Є–Љ–Є –Є—Б–њ–Њ–ї—М–Ј–Њ–≤–∞—В—М –Є—Е –і–ї—П —А–µ—И–µ–љ–Є—П —В–µ—Е –Є–ї–Є –Є–љ—Л—Е –Ј–∞–і–∞—З. –Ф–∞–≤–∞–є—В–µ —Б—А–∞–≤–љ–Є–Љ –Є —Б–Њ–њ–Њ—Б—В–∞–≤–Є–Љ –љ–µ–Ї–Њ—В–Њ—А—Л–µ —А–∞—Б–њ—А–Њ—Б—В—А–∞–љ–µ–љ–љ—Л–µ —Б–µ—В–µ–≤—Л–µ —Г—Б—В—А–Њ–є—Б—В–≤–∞ –Є –Њ–±—Б—Г–і–Є–Љ –Є—Е –њ—А–∞–≤–Є–ї—М–љ–Њ–µ —А–∞–Ј–Љ–µ—Й–µ–љ–Є–µ:

  • –†–Њ—Г—В–µ—А—Л: –Ь–∞—А—И—А—Г—В–Є–Ј–∞—В–Њ—А—Л –Њ—В–≤–µ—З–∞—О—В –Ј–∞ –љ–∞–њ—А–∞–≤–ї–µ–љ–Є–µ —В—А–∞—Д–Є–Ї–∞ –Љ–µ–ґ–і—Г —А–∞–Ј–ї–Є—З–љ—Л–Љ–Є —Б–µ—В—П–Љ–Є. –Ю–љ–Є —А–∞–±–Њ—В–∞—О—В –љ–∞ —Б–µ—В–µ–≤–Њ–Љ —Г—А–Њ–≤–љ–µ (Layer 3) –Є –Є—Б–њ–Њ–ї—М–Ј—Г—О—В –њ—А–Њ—В–Њ–Ї–Њ–ї—Л –Љ–∞—А—И—А—Г—В–Є–Ј–∞—Ж–Є–Є –і–ї—П –Њ–њ—А–µ–і–µ–ї–µ–љ–Є—П –Њ–њ—В–Є–Љ–∞–ї—М–љ–Њ–≥–Њ –њ—Г—В–Є –њ–µ—А–µ–і–∞—З–Є –і–∞–љ–љ—Л—Е.

  • –Ъ–Њ–Љ–Љ—Г—В–∞—В–Њ—А—Л**: –Ъ–Њ–Љ–Љ—Г—В–∞—В–Њ—А—Л —А–∞–±–Њ—В–∞—О—В –љ–∞ –Ї–∞–љ–∞–ї—М–љ–Њ–Љ —Г—А–Њ–≤–љ–µ (Layer 2) –Є –Њ–±–µ—Б–њ–µ—З–Є–≤–∞—О—В —Б–≤—П–Ј—М –Љ–µ–ґ–і—Г —Г—Б—В—А–Њ–є—Б—В–≤–∞–Љ–Є –≤ –ї–Њ–Ї–∞–ї—М–љ–Њ–є —Б–µ—В–Є (LAN). –Ю–љ–Є –Є—Б–њ–Њ–ї—М–Ј—Г—О—В MAC-–∞–і—А–µ—Б–∞ –і–ї—П –њ–µ—А–µ—Б—Л–ї–Ї–Є –њ–∞–Ї–µ—В–Њ–≤ –і–∞–љ–љ—Л—Е –∞–і—А–µ—Б–∞—В–∞–Љ.

  • –С—А–∞–љ–і–Љ–∞—Г—Н—А—Л: –Ь–µ–ґ—Б–µ—В–µ–≤—Л–µ —Н–Ї—А–∞–љ—Л –Ј–∞—Й–Є—Й–∞—О—В —Б–µ—В–Є –Њ—В –љ–µ—Б–∞–љ–Ї—Ж–Є–Њ–љ–Є—А–Њ–≤–∞–љ–љ–Њ–≥–Њ –і–Њ—Б—В—Г–њ–∞ –Є –≤—А–µ–і–Њ–љ–Њ—Б–љ–Њ–≥–Њ —В—А–∞—Д–Є–Ї–∞. –Ю–љ–Є –Њ–±–µ—Б–њ–µ—З–Є–≤–∞—О—В —Б–Њ–±–ї—О–і–µ–љ–Є–µ –њ–Њ–ї–Є—В–Є–Ї –±–µ–Ј–Њ–њ–∞—Б–љ–Њ—Б—В–Є, –њ—А–Њ–≤–µ—А—П—П —Б–µ—В–µ–≤–Њ–є —В—А–∞—Д–Є–Ї –Є —А–∞–Ј—А–µ—И–∞—П –Є–ї–Є –±–ї–Њ–Ї–Є—А—Г—П –Њ–њ—А–µ–і–µ–ї–µ–љ–љ—Л–µ —Б–Њ–µ–і–Є–љ–µ–љ–Є—П –љ–∞ –Њ—Б–љ–Њ–≤–µ –Ј–∞–і–∞–љ–љ—Л—Е –њ—А–∞–≤–Є–ї.

  • –С–∞–ї–∞–љ—Б–Є—А–Њ–≤—Й–Є–Ї–Є –љ–∞–≥—А—Г–Ј–Ї–Є: –С–∞–ї–∞–љ—Б–Є—А–Њ–≤—Й–Є–Ї–Є –љ–∞–≥—А—Г–Ј–Ї–Є —А–∞—Б–њ—А–µ–і–µ–ї—П—О—В –≤—Е–Њ–і—П—Й–Є–є —Б–µ—В–µ–≤–Њ–є —В—А–∞—Д–Є–Ї –Љ–µ–ґ–і—Г –љ–µ—Б–Ї–Њ–ї—М–Ї–Є–Љ–Є —Б–µ—А–≤–µ—А–∞–Љ–Є –і–ї—П –Њ–њ—В–Є–Љ–Є–Ј–∞—Ж–Є–Є –Є—Б–њ–Њ–ї—М–Ј–Њ–≤–∞–љ–Є—П —А–µ—Б—Г—А—Б–Њ–≤, –њ–Њ–≤—Л—И–µ–љ–Є—П –њ—А–Њ–Є–Ј–≤–Њ–і–Є—В–µ–ї—М–љ–Њ—Б—В–Є –Є –Њ–±–µ—Б–њ–µ—З–µ–љ–Є—П –≤—Л—Б–Њ–Ї–Њ–є –і–Њ—Б—В—Г–њ–љ–Њ—Б—В–Є.

  • –Ґ–Њ—З–Ї–Є –і–Њ—Б—В—Г–њ–∞: –Ґ–Њ—З–Ї–Є –і–Њ—Б—В—Г–њ–∞ (AP) –Њ–±–µ—Б–њ–µ—З–Є–≤–∞—О—В –±–µ—Б–њ—А–Њ–≤–Њ–і–љ–Њ–µ –њ–Њ–і–Ї–ї—О—З–µ–љ–Є–µ —Г—Б—В—А–Њ–є—Б—В–≤ –≤ —Б–µ—В–Є. –Ю–љ–Є –Њ–±–µ—Б–њ–µ—З–Є–≤–∞—О—В –±–µ—Б–њ—А–Њ–≤–Њ–і–љ—Г—О —Б–≤—П–Ј—М, –њ–µ—А–µ–і–∞–≤–∞—П –Є –њ—А–Є–љ–Є–Љ–∞—П –і–∞–љ–љ—Л–µ –Љ–µ–ґ–і—Г –±–µ—Б–њ—А–Њ–≤–Њ–і–љ—Л–Љ–Є —Г—Б—В—А–Њ–є—Б—В–≤–∞–Љ–Є –Є –њ—А–Њ–≤–Њ–і–љ–Њ–є —Б–µ—В—М—О.
