–Т –Њ—В–ї–Є—З–Є–µ –Њ—В —В—А–∞–і–Є—Ж–Є–Њ–љ–љ—Л—Е –Љ–µ—Е–∞–љ–Є–Ј–Љ–Њ–≤ PoW, —В—А–µ–±—Г—О—Й–Є—Е –Њ–≥—А–Њ–Љ–љ—Л—Е –≤—Л—З–Є—Б–ї–Є—В–µ–ї—М–љ—Л—Е –Љ–Њ—Й–љ–Њ—Б—В–µ–є –Є —Н–љ–µ—А–≥–Њ–њ–Њ—В—А–µ–±–ї–µ–љ–Є—П, Spacemesh-–Љ–∞–є–љ–Є–љ–≥ –±–Њ–ї–µ–µ —Н–Ї–Њ–ї–Њ–≥–Є—З–µ–љ. –Ю–љ –њ—А–µ–і–њ–Њ–ї–∞–≥–∞–µ—В –Є—Б–њ–Њ–ї—М–Ј–Њ–≤–∞–љ–Є–µ —Б—Г—Й–µ—Б—В–≤—Г—О—Й–µ–≥–Њ –Њ–±–Њ—А—Г–і–Њ–≤–∞–љ–Є—П –Є —Б–Є—Б—В–µ–Љ —Е—А–∞–љ–µ–љ–Є—П –і–∞–љ–љ—Л—Е, —З—В–Њ –Ј–љ–∞—З–Є—В–µ–ї—М–љ–Њ —Б–љ–Є–ґ–∞–µ—В –µ–≥–Њ “—Г–≥–ї–µ—А–Њ–і–љ—Л–є —Б–ї–µ–і”.

2. –Ф–µ—Ж–µ–љ—В—А–∞–ї–Є–Ј–∞—Ж–Є—П #

–Р–ї–≥–Њ—А–Є—В–Љ –Ї–Њ–љ—Б–µ–љ—Б—Г—Б–∞ PoST –≤ Spacemesh —Б–њ–Њ—Б–Њ–±—Б—В–≤—Г–µ—В –і–µ—Ж–µ–љ—В—А–∞–ї–Є–Ј–∞—Ж–Є–Є, –њ–Њ–Ј–≤–Њ–ї—П—П —Г—З–∞—Б—В–≤–Њ–≤–∞—В—М –≤ –љ–µ–Љ –≤—Б–µ–Љ, —Г –Ї–Њ–≥–Њ –µ—Б—В—М –Љ–µ—Б—В–Њ –і–ї—П —Е—А–∞–љ–µ–љ–Є—П –і–∞–љ–љ—Л—Е. –≠—В–Њ –њ—А–µ–і–Њ—В–≤—А–∞—Й–∞–µ—В —Ж–µ–љ—В—А–∞–ї–Є–Ј–∞—Ж–Є—О –Љ–∞–є–љ–Є–љ–≥–∞ –≤ —А—Г–Ї–∞—Е –љ–µ—Б–Ї–Њ–ї—М–Ї–Є—Е –≤–ї–Є—П—В–µ–ї—М–љ—Л—Е –Њ—А–≥–∞–љ–Є–Ј–∞—Ж–Є–є, —З—В–Њ –њ—А–Є–≤–Њ–і–Є—В –Ї –±–Њ–ї–µ–µ —Б–њ—А–∞–≤–µ–і–ї–Є–≤–Њ–Љ—Г —А–∞—Б–њ—А–µ–і–µ–ї–µ–љ–Є—О –≤–Њ–Ј–љ–∞–≥—А–∞–ґ–і–µ–љ–Є–є –Є –њ–Њ–≤—Л—И–µ–љ–Є—О –±–µ–Ј–Њ–њ–∞—Б–љ–Њ—Б—В–Є —Б–µ—В–Є.

3. –С–µ–Ј–Њ–њ–∞—Б–љ–Њ—Б—В—М –Є —Н—Д—Д–µ–Ї—В–Є–≤–љ–Њ—Б—В—М #

–Р–ї–≥–Њ—А–Є—В–Љ PoST –Њ–±–µ—Б–њ–µ—З–Є–≤–∞–µ—В –≤—Л—Б–Њ–Ї–Є–є —Г—А–Њ–≤–µ–љ—М –±–µ–Ј–Њ–њ–∞—Б–љ–Њ—Б—В–Є –Є —Н—Д—Д–µ–Ї—В–Є–≤–љ–Њ—Б—В–Є —Б–µ—В–Є. –Ь–µ—Е–∞–љ–Є–Ј–Љ –Ї–Њ–љ—Б–µ–љ—Б—Г—Б–∞ —В—А–µ–±—Г–µ—В –Њ—В –і–Њ–±—А–Њ—Б–Њ–≤–µ—Б—В–љ—Л—Е mesh-—Б–Є—Б—В–µ–Љ –≤—Л–і–µ–ї–µ–љ–Є—П –і–Є—Б–Ї–Њ–≤–Њ–≥–Њ –њ—А–Њ—Б—В—А–∞–љ—Б—В–≤–∞ –љ–∞ –і–ї–Є—В–µ–ї—М–љ—Л–є —Б—А–Њ–Ї, —З—В–Њ –і–µ–ї–∞–µ—В –љ–µ–≤–Њ–Ј–Љ–Њ–ґ–љ—Л–Љ –њ–Њ–ї—Г—З–µ–љ–Є–µ –Ї–Њ–љ—В—А–Њ–ї—П –љ–∞–і —Б–µ—В—М—О –Ј–ї–Њ—Г–Љ—Л—И–ї–µ–љ–љ–Є–Ї–∞–Љ–Є.

–Я–Њ—В–µ–љ—Ж–Є–∞–ї—М–љ—Л–µ –њ—А–Њ–±–ї–µ–Љ—Л –Є –Њ–њ–∞—Б–µ–љ–Є—П #

1. –Ґ—А–µ–±–Њ–≤–∞–љ–Є—П –Ї —Е—А–∞–љ–µ–љ–Є—О –і–∞–љ–љ—Л—Е #

Spacemesh-–Љ–∞–є–љ–Є–љ–≥ —В—А–µ–±—Г–µ—В –Ј–љ–∞—З–Є—В–µ–ї—М–љ–Њ–≥–Њ –Њ–±—К–µ–Љ–∞ –і–Є—Б–Ї–Њ–≤–Њ–≥–Њ –њ—А–Њ—Б—В—А–∞–љ—Б—В–≤–∞. –Я–Њ –Љ–µ—А–µ —А–Њ—Б—В–∞ —Б–µ—В–Є —В—А–µ–±–Њ–≤–∞–љ–Є—П –Ї —Е—А–∞–љ–µ–љ–Є—О –і–∞–љ–љ—Л—Е –Љ–Њ–≥—Г—В —Г–≤–µ–ї–Є—З–Є–≤–∞—В—М—Б—П, —З—В–Њ –Љ–Њ–ґ–µ—В –Њ–≥—А–∞–љ–Є—З–Є—В—М —Г—З–∞—Б—В–Є–µ –≤ –њ—А–Њ–µ–Ї—В–µ —В–µ—Е, –Ї—В–Њ —А–∞—Б–њ–Њ–ї–∞–≥–∞–µ—В –і–Њ—Б—В–∞—В–Њ—З–љ—Л–Љ–Є —А–µ—Б—Г—А—Б–∞–Љ–Є –і–ї—П —Е—А–∞–љ–µ–љ–Є—П –і–∞–љ–љ—Л—Е.

2. –°–ї–Њ–ґ–љ–Њ—Б—В—М –њ–Њ—Б—В—А–Њ–µ–љ–Є—П —Б–µ—В–Ї–Є #

–Я—А–Њ—Ж–µ—Б—Б –і–Є–љ–∞–Љ–Є—З–µ—Б–Ї–Њ–≥–Њ –Ј–∞—Ж–µ–њ–ї–µ–љ–Є—П, —Е–Њ—В—П –Є —П–≤–ї—П–µ—В—Б—П –≤–∞–ґ–љ—Л–Љ –і–ї—П –і–µ—Ж–µ–љ—В—А–∞–ї–Є–Ј–∞—Ж–Є–Є, –Љ–Њ–ґ–µ—В —Б–Њ–Ј–і–∞–≤–∞—В—М –њ—А–Њ–±–ї–µ–Љ—Л, —Б–≤—П–Ј–∞–љ–љ—Л–µ —Б –Ї–Њ–Љ–Љ—Г–љ–Є–Ї–∞—Ж–Є–µ–є –Є –Ї–Њ–Њ—А–і–Є–љ–∞—Ж–Є–µ–є –Љ–µ–ґ–і—Г –Ј–∞—Ж–µ–њ–ї—П—О—Й–Є–Љ–Є —Г—Б—В—А–Њ–є—Б—В–≤–∞–Љ–Є. –Ю–њ—В–Є–Љ–Є–Ј–∞—Ж–Є—П —Н—В–Њ–≥–Њ –њ—А–Њ—Ж–µ—Б—Б–∞ –Є–Љ–µ–µ—В —А–µ—И–∞—О—Й–µ–µ –Ј–љ–∞—З–µ–љ–Є–µ –і–ї—П –Љ–∞—Б—И—В–∞–±–Є—А—Г–µ–Љ–Њ—Б—В–Є —Б–µ—В–Є.

3. –Ъ–Њ–љ–Ї—Г—А–µ–љ—Ж–Є—П –Є –њ—А–Є–±—Л–ї—М–љ–Њ—Б—В—М #

–Я–Њ –Љ–µ—А–µ —Г–≤–µ–ї–Є—З–µ–љ–Є—П —З–Є—Б–ї–∞ —Г—З–∞—Б—В–љ–Є–Ї–Њ–≤ —Б–µ—В–Є Spacemesh –Ї–Њ–љ–Ї—Г—А–µ–љ—Ж–Є—П –Ј–∞ –≤–Њ–Ј–љ–∞–≥—А–∞–ґ–і–µ–љ–Є–µ –Љ–Њ–ґ–µ—В —Г—Б–Є–ї–Є–≤–∞—В—М—Б—П. –Ь–µ—И–µ—А—Л –і–Њ–ї–ґ–љ—Л —В—Й–∞—В–µ–ї—М–љ–Њ –Њ—Ж–µ–љ–Є–≤–∞—В—М –Ј–∞—В—А–∞—В—Л –љ–∞ —Е—А–∞–љ–µ–љ–Є–µ –і–∞–љ–љ—Л—Е, —Н–ї–µ–Ї—В—А–Њ—Н–љ–µ—А–≥–Є—О –Є –Њ–±–Њ—А—Г–і–Њ–≤–∞–љ–Є–µ –≤ —Б—А–∞–≤–љ–µ–љ–Є–Є —Б –њ–Њ—В–µ–љ—Ж–Є–∞–ї—М–љ—Л–Љ –і–Њ—Е–Њ–і–Њ–Љ, —З—В–Њ–±—Л –Љ–∞–є–љ–Є–љ–≥ –Њ—Б—В–∞–≤–∞–ї—Б—П –њ—А–Є–±—Л–ї—М–љ—Л–Љ.

–Ч–∞–Ї–ї—О—З–µ–љ–Є–µ #

–Ь–∞–є–љ–Є–љ–≥ Spacemesh –њ—А–µ–і—Б—В–∞–≤–ї—П–µ—В —Б–Њ–±–Њ–є –љ–Њ–≤–∞—В–Њ—А—Б–Ї–Є–є –њ–Њ–і—Е–Њ–і –Ї –Ї—А–Є–њ—В–Њ–≤–∞–ї—О—В–љ–Њ–Љ—Г –Ї–Њ–љ—Б–µ–љ—Б—Г—Б—Г, –≤ –Ї–Њ—В–Њ—А–Њ–Љ –њ—А–Є–Њ—А–Є—В–µ—В–љ—Л–Љ–Є —П–≤–ї—П—О—В—Б—П —Н–Ї–Њ–ї–Њ–≥–Є—З–µ—Б–Ї–∞—П —Г—Б—В–Њ–є—З–Є–≤–Њ—Б—В—М, –і–µ—Ж–µ–љ—В—А–∞–ї–Є–Ј–∞—Ж–Є—П –Є –±–µ–Ј–Њ–њ–∞—Б–љ–Њ—Б—В—М —Б–µ—В–Є. –Ш—Б–њ–Њ–ї—М–Ј—Г—П –∞–ї–≥–Њ—А–Є—В–Љ PoST, Spacemesh –і–Њ–±–Є–≤–∞–µ—В—Б—П —Н–Ї–Њ–ї–Њ–≥–Є—З–љ–Њ–≥–Њ –Є —Б–њ—А–∞–≤–µ–і–ї–Є–≤–Њ–≥–Њ —А–∞—Б–њ—А–µ–і–µ–ї–µ–љ–Є—П –≤–Њ–Ј–љ–∞–≥—А–∞–ґ–і–µ–љ–Є—П –Ј–∞ –Љ–∞–є–љ–Є–љ–≥, –Њ–±–µ—Б–њ–µ—З–Є–≤–∞—П –њ—А–Є —Н—В–Њ–Љ —Б—В–∞–±–Є–ї—М–љ–Њ—Б—В—М –Є —Н—Д—Д–µ–Ї—В–Є–≤–љ–Њ—Б—В—М —Б–µ—В–Є.

–Я–Њ –Љ–µ—А–µ —А–∞–Ј–≤–Є—В–Є—П –Ї—А–Є–њ—В–Њ–≤–∞–ї—О—В–љ–Њ–≥–Њ –ї–∞–љ–і—И–∞—Д—В–∞ –Є–љ–љ–Њ–≤–∞—Ж–Є–Њ–љ–љ—Л–є –њ–Њ–і—Е–Њ–і Spacemesh –Љ–Њ–ґ–µ—В –њ—А–Њ–ї–Њ–ґ–Є—В—М –њ—Г—В—М –Ї –±–Њ–ї–µ–µ —Г—Б—В–Њ–є—З–Є–≤—Л–Љ –Є –Є–љ–Ї–ї—О–Ј–Є–≤–љ—Л–Љ –њ—А–∞–Ї—В–Є–Ї–∞–Љ –Љ–∞–є–љ–Є–љ–≥–∞. –Э–µ–Ј–∞–≤–Є—Б–Є–Љ–Њ –Њ—В —В–Њ–≥–Њ, —П–≤–ї—П–µ—В–µ—Б—М –ї–Є –≤—Л –њ–Њ—В–µ–љ—Ж–Є–∞–ї—М–љ—Л–Љ –Љ–∞–є–љ–µ—А–Њ–Љ –Є–ї–Є –њ—А–Њ—Б—В–Њ –Є–љ—В–µ—А–µ—Б—Г–µ—В–µ—Б—М –њ–Њ—Б–ї–µ–і–љ–Є–Љ–Є —Б–Њ–±—Л—В–Є—П–Љ–Є –≤ –Љ–Є—А–µ –Ї—А–Є–њ—В–Њ–≤–∞–ї—О—В, –Љ–∞–є–љ–Є–љ–≥ Spacemesh, –љ–µ—Б–Њ–Љ–љ–µ–љ–љ–Њ, —П–≤–ї—П–µ—В—Б—П –Є–љ—В–µ—А–µ—Б–љ—Л–Љ –љ–∞–њ—А–∞–≤–ї–µ–љ–Є–µ–Љ.
